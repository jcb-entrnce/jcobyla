34  /**
35   * Constrained Optimization BY Linear Approximation in Java.
36   *
37   * COBYLA2 is an implementation of Powell's nonlinear derivative free
38   * constrained optimization that uses a linear approximation approach. The
39   * algorithm is a sequential trusted region algorithm that employs linear
40   * approximations to the objective and constraint functions, where the
41   * approximations are formed by linear interpolation at n + 1 points in the
42   * space of the variables and tries to maintain a regular shaped simplex over
43   * iterations.
44   *
45   * It solves nonsmooth NLP with a moderate number of variables (about 100).
46   * Inequality constraints only.
47   *
48   * The initial point X is taken as one vertex of the initial simplex with zero
49   * being another, so, X should not be entered as the zero vector.
50   *
51   * @author Anders Gustafsson, Cureos AB.
52   */
53  public class Cobyla {
54         //     provide an initial vector of variables in X.  Further, the value of
55          //     IPRINT should be set to 0, 1, 2 or 3, which controls the amount of
56          //     printing during the calculation. Specifically, there is no output if
57          //     IPRINT=0 and there is output only at the end of the calculation if
58          //     IPRINT=1.  Otherwise each new value of RHO and SIGMA is printed.
59          //     Further, the vector of variables and some function information are
60          //     given either when RHO is reduced or when each new value of F(X) is
61          //     computed in the cases IPRINT=2 or IPRINT=3 respectively.    
62      /**
63       * No output.
64       */
65      public final static int IPRINT_NONE = 0;
66      /**
67       * Output only at the end of the calculation.
68       */
69      public final static int IPRINT_END_OF_CALC = 1;
70      /**
71       * Output new values of RHO and SIGMA, and furthermore X when RHO is reduced.
72       */
73      public final static int IPRINT_RHO_REDUCED = 2;
74      /**
75       * Output new values of RHO and SIGMA, and furthermore X for each new value
76       * of F(X).
77       */
78      public final static int IPRINT_FULL = 3;
79  
80      /**
81       * Minimizes the objective function F with respect to a set of inequality
82       * constraints CON, and returns the optimal variable array. F and CON may be
83       * non-linear, and should preferably be smooth.
84       *
85       * @param calcfc Interface implementation for calculating objective function
86       * and constraints.
87       * @param n Number of variables.
88       * @param m Number of constraints.
89       * @param x On input initial values of the variables (zero-based array). On
90       * output optimal values of the variables obtained in the COBYLA
91       * minimization.
92       * @param rhobeg Initial size of the simplex.
93       * @param rhoend Final value of the simplex.
94       * @param iprint Print level, 0 &lt;= iprint &lt;= 3, where 0 provides no
95       * output and 3 provides full output to the console.
96       * @param maxfun Maximum number of function evaluations before terminating.
97       * @return Exit status of the COBYLA2 optimization.
98       */
99      public static CobylaExitStatus findMinimum(final Calcfc calcfc, int n,
100             int m, double[] x, double rhobeg, double rhoend, int iprint,
101             int maxfun) {
102         //     This subroutine minimizes an objective function F(X) subject to M
103         //     inequality constraints on X, where X is a vector of variables that has
104         //     N components.  The algorithm employs linear approximations to the
105         //     objective and constraint functions, the approximations being formed by
106         //     linear interpolation at N+1 points in the space of the variables.
107         //     We regard these interpolation points as vertices of a simplex.  The
108         //     parameter RHO controls the size of the simplex and it is reduced
109         //     automatically from RHOBEG to RHOEND.  For each RHO the subroutine tries
110         //     to achieve a good vector of variables for the current size, and then
111         //     RHO is reduced until the value RHOEND is reached.  Therefore RHOBEG and
112         //     RHOEND should be set to reasonable initial changes to and the required
113         //     accuracy in the variables respectively, but this accuracy should be
114         //     viewed as a subject for experimentation because it is not guaranteed.
115         //     The subroutine has an advantage over many of its competitors, however,
116         //     which is that it treats each constraint individually when calculating
117         //     a change to the variables, instead of lumping the constraints together
118         //     into a single penalty function.  The name of the subroutine is derived
119         //     from the phrase Constrained Optimization BY Linear Approximations.
120 
121         //     The user must set the values of N, M, RHOBEG and RHOEND, and must
122         //     provide an initial vector of variables in X.  Further, the value of
123         //     IPRINT should be set to 0, 1, 2 or 3, which controls the amount of
124         //     printing during the calculation. Specifically, there is no output if
125         //     IPRINT=0 and there is output only at the end of the calculation if
126         //     IPRINT=1.  Otherwise each new value of RHO and SIGMA is printed.
127         //     Further, the vector of variables and some function information are
128         //     given either when RHO is reduced or when each new value of F(X) is
129         //     computed in the cases IPRINT=2 or IPRINT=3 respectively. Here SIGMA
130         //     is a penalty parameter, it being assumed that a change to X is an
131         //     improvement if it reduces the merit function
132         //                F(X)+SIGMA*MAX(0.0, - C1(X), - C2(X),..., - CM(X)),
133         //     where C1,C2,...,CM denote the constraint functions that should become
134         //     nonnegative eventually, at least to the precision of RHOEND. In the
135         //     printed output the displayed term that is multiplied by SIGMA is
136         //     called MAXCV, which stands for 'MAXimum Constraint Violation'.  The
137         //     argument ITERS is an integer variable that must be set by the user to a
138         //     limit on the number of calls of CALCFC, the purpose of this routine being
139         //     given below.  The value of ITERS will be altered to the number of calls
140         //     of CALCFC that are made.
141         //     In order to define the objective and constraint functions, we require
142         //     a subroutine that has the name and arguments
143         //                SUBROUTINE CALCFC (N,M,X,F,CON)
144         //                DIMENSION X(:),CON(:)  .
145         //     The values of N and M are fixed and have been defined already, while
146         //     X is now the current vector of variables. The subroutine should return
147         //     the objective and constraint functions at X in F and CON(1),CON(2),
148         //     ...,CON(M).  Note that we are trying to adjust X so that F(X) is as
149         //     small as possible subject to the constraint functions being nonnegative.
150         // Local variables
151         int mpp = m + 2;
152 
153         // Internal base-1 X array
154         double[] iox = new double[n + 1];
155         System.arraycopy(x, 0, iox, 1, n);
156 
157         // Internal representation of the objective and constraints calculation method, 
158         // accounting for that X and CON arrays in the cobylb method are base-1 arrays.
159         Calcfc fcalcfc = new Calcfc() {
160             /**
161              *
162              * @param n the value of n
163              * @param m the value of m
164              * @param x the value of x
165              * @param con the value of con
166              * @return the double
167              */
168             @Override
169             public double compute(int n, int m, double[] x, double[] con) {
170                 double[] ix = new double[n];
171                 System.arraycopy(x, 1, ix, 0, n);
172                 double[] ocon = new double[m];
173                 double f = calcfc.compute(n, m, ix, ocon);
174                 System.arraycopy(ocon, 0, con, 1, m);
175                 return f;
176             }
177         };
178 
179         CobylaExitStatus status = cobylb(fcalcfc, n, m, mpp, iox, rhobeg, rhoend, iprint, maxfun);
180         System.arraycopy(iox, 1, x, 0, n);
181 
182         return status;
183     }
184 
185     private static CobylaExitStatus cobylb(Calcfc calcfc, int n, int m, int mpp, double[] x,
186             double rhobeg, double rhoend, int iprint, int maxfun) {
187         // N.B. Arguments CON, SIM, SIMI, DATMAT, A, VSIG, VETA, SIGBAR, DX, W & IACT
188         //      have been removed.
189 
190         //     Set the initial values of some parameters. The last column of SIM holds
191         //     the optimal vertex of the current simplex, and the preceding N columns
192         //     hold the displacements from the optimal vertex to the other vertices.
193         //     Further, SIMI holds the inverse of the matrix that is contained in the
194         //     first N columns of SIM.
195         // Local variables
196         CobylaExitStatus status;
197 
198         double alpha = 0.25;
199         double beta = 2.1;
200         double gamma = 0.5;
201         double delta = 1.1;
202 
203         double f = 0.0;
204         double resmax = 0.0;
205         double total;
206 
207         int np = n + 1;
208         int mp = m + 1;
209         double rho = rhobeg;
210         double parmu = 0.0;
211 
212         boolean iflag = false;
213         boolean ifull = false;
214         double parsig = 0.0;
215         double prerec = 0.0;
216         double prerem = 0.0;
217 
218         double[] con = new double[1 + mpp];
219         double[][] sim = new double[1 + n][1 + np];
220         double[][] simi = new double[1 + n][1 + n];
221         double[][] datmat = new double[1 + mpp][1 + np];
222         double[][] a = new double[1 + n][1 + mp];
223         double[] vsig = new double[1 + n];
224         double[] veta = new double[1 + n];
225         double[] sigbar = new double[1 + n];
226         double[] dx = new double[1 + n];
227         double[] w = new double[1 + n];
228 
229         if (iprint >= 2) {
230             System.out.format("%nThe initial value of RHO is %13.6f and PARMU is set to zero.%n", rho);
231         }
232 
233         int nfvals = 0;
234         double temp = 1.0 / rho;
235 
236         for (int i = 1; i <= n; ++i) {
237             sim[i][np] = x[i];
238             sim[i][i] = rho;
239             simi[i][i] = temp;
240         }
241 
242         int jdrop = np;
243         boolean ibrnch = false;
244 
245         //     Make the next call of the user-supplied subroutine CALCFC. These
246         //     instructions are also used for calling CALCFC during the iterations of
247         //     the algorithm.
248         L_40:
249         do {
250             if (nfvals >= maxfun && nfvals > 0) {
251                 status = CobylaExitStatus.MAX_ITERATIONS_REACHED;
252                 break L_40;
253             }
254 
255             ++nfvals;
256 
257             f = calcfc.compute(n, m, x, con);
258             resmax = 0.0;
259             for (int k = 1; k <= m; ++k) {
260                 resmax = Math.max(resmax, -con[k]);
261             }
262 
263             if (nfvals == iprint - 1 || iprint == 3) {
264                 printIterationResult(nfvals, f, resmax, x, n);
265             }
266 
267             con[mp] = f;
268             con[mpp] = resmax;
269 
270             //     Set the recently calculated function values in a column of DATMAT. This
271             //     array has a column for each vertex of the current simplex, the entries of
272             //     each column being the values of the constraint functions (if any)
273             //     followed by the objective function and the greatest constraint violation
274             //     at the vertex.
275             boolean skipVertexIdent = true;
276             if (!ibrnch) {
277                 skipVertexIdent = false;
278 
279                 for (int i = 1; i <= mpp; ++i) {
280                     datmat[i][jdrop] = con[i];
281                 }
282 
283                 if (nfvals <= np) {
284                     //     Exchange the new vertex of the initial simplex with the optimal vertex if
285                     //     necessary. Then, if the initial simplex is not complete, pick its next
286                     //     vertex and calculate the function values there.
287 
288                     if (jdrop <= n) {
289                         if (datmat[mp][np] <= f) {
290                             x[jdrop] = sim[jdrop][np];
291                         } else {
292                             sim[jdrop][np] = x[jdrop];
293                             for (int k = 1; k <= mpp; ++k) {
294                                 datmat[k][jdrop] = datmat[k][np];
295                                 datmat[k][np] = con[k];
296                             }
297                             for (int k = 1; k <= jdrop; ++k) {
298                                 sim[jdrop][k] = -rho;
299                                 temp = 0.0;
300                                 for (int i = k; i <= jdrop; ++i) {
301                                     temp -= simi[i][k];
302                                 }
303                                 simi[jdrop][k] = temp;
304                             }
305                         }
306                     }
307                     if (nfvals <= n) {
308                         jdrop = nfvals;
309                         x[jdrop] += rho;
310                         continue L_40;
311                     }
312                 }
313 
314                 ibrnch = true;
315             }
316 
317             L_140:
318             do {
319                 L_550:
320                 do {
321                     if (!skipVertexIdent) {
322                         //     Identify the optimal vertex of the current simplex.
323 
324                         double phimin = datmat[mp][np] + parmu * datmat[mpp][np];
325                         int nbest = np;
326 
327                         for (int j = 1; j <= n; ++j) {
328                             temp = datmat[mp][j] + parmu * datmat[mpp][j];
329                             if (temp < phimin) {
330                                 nbest = j;
331                                 phimin = temp;
332                             } else if (temp == phimin && parmu == 0.0 && datmat[mpp][j] < datmat[mpp][nbest]) {
333                                 nbest = j;
334                             }
335                         }
336 
337                         //     Switch the best vertex into pole position if it is not there already,
338                         //     and also update SIM, SIMI and DATMAT.
339                         if (nbest <= n) {
340                             for (int i = 1; i <= mpp; ++i) {
341                                 temp = datmat[i][np];
342                                 datmat[i][np] = datmat[i][nbest];
343                                 datmat[i][nbest] = temp;
344                             }
345                             for (int i = 1; i <= n; ++i) {
346                                 temp = sim[i][nbest];
347                                 sim[i][nbest] = 0.0;
348                                 sim[i][np] += temp;
349 
350                                 double tempa = 0.0;
351                                 for (int k = 1; k <= n; ++k) {
352                                     sim[i][k] -= temp;
353                                     tempa -= simi[k][i];
354                                 }
355                                 simi[nbest][i] = tempa;
356                             }
357                         }
358 
359                         //     Make an error return if SIGI is a poor approximation to the inverse of
360                         //     the leading N by N submatrix of SIG.
361                         double error = 0.0;
362                         for (int i = 1; i <= n; ++i) {
363                             for (int j = 1; j <= n; ++j) {
364                                 temp = dotProduct(part(row(simi, i), 1, n), part(col(sim, j), 1, n)) - (i == j ? 1.0 : 0.0);
365                                 error = Math.max(error, Math.abs(temp));
366                             }
367                         }
368                         if (error > 0.1) {
369                             status = CobylaExitStatus.DIVERGING_ROUNDING_ERRORS;
370                             break L_40;
371                         }
372 
373                         //     Calculate the coefficients of the linear approximations to the objective
374                         //     and constraint functions, placing minus the objective function gradient
375                         //     after the constraint gradients in the array A. The vector W is used for
376                         //     working space.
377                         for (int k = 1; k <= mp; ++k) {
378                             con[k] = -datmat[k][np];
379                             for (int j = 1; j <= n; ++j) {
380                                 w[j] = datmat[k][j] + con[k];
381                             }
382 
383                             for (int i = 1; i <= n; ++i) {
384                                 a[i][k] = (k == mp ? -1.0 : 1.0) * dotProduct(part(w, 1, n), part(col(simi, i), 1, n));
385                             }
386                         }
387 
388                         //     Calculate the values of sigma and eta, and set IFLAG = 0 if the current
389                         //     simplex is not acceptable.
390                         iflag = true;
391                         parsig = alpha * rho;
392                         double pareta = beta * rho;
393 
394                         for (int j = 1; j <= n; ++j) {
395                             double wsig = 0.0;
396                             for (int k = 1; k <= n; ++k) {
397                                 wsig += simi[j][k] * simi[j][k];
398                             }
399                             double weta = 0.0;
400                             for (int k = 1; k <= n; ++k) {
401                                 weta += sim[k][j] * sim[k][j];
402                             }
403                             vsig[j] = 1.0 / Math.sqrt(wsig);
404                             veta[j] = Math.sqrt(weta);
405                             if (vsig[j] < parsig || veta[j] > pareta) {
406                                 iflag = false;
407                             }
408                         }
409 
410                         //     If a new vertex is needed to improve acceptability, then decide which
411                         //     vertex to drop from the simplex.
412                         if (!ibrnch && !iflag) {
413                             jdrop = 0;
414                             temp = pareta;
415                             for (int j = 1; j <= n; ++j) {
416                                 if (veta[j] > temp) {
417                                     jdrop = j;
418                                     temp = veta[j];
419                                 }
420                             }
421                             if (jdrop == 0) {
422                                 for (int j = 1; j <= n; ++j) {
423                                     if (vsig[j] < temp) {
424                                         jdrop = j;
425                                         temp = vsig[j];
426                                     }
427                                 }
428                             }
429 
430                             //     Calculate the step to the new vertex and its sign.
431                             temp = gamma * rho * vsig[jdrop];
432                             for (int k = 1; k <= n; ++k) {
433                                 dx[k] = temp * simi[jdrop][k];
434                             }
435                             double cvmaxp = 0.0;
436                             double cvmaxm = 0.0;
437 
438                             total = 0.0;
439                             for (int k = 1; k <= mp; ++k) {
440                                 total = dotProduct(part(col(a, k), 1, n), part(dx, 1, n));
441                                 if (k < mp) {
442                                     temp = datmat[k][np];
443                                     cvmaxp = Math.max(cvmaxp, -total - temp);
444                                     cvmaxm = Math.max(cvmaxm, total - temp);
445                                 }
446                             }
447                             double dxsign = parmu * (cvmaxp - cvmaxm) > 2.0 * total ? -1.0 : 1.0;
448 
449                             //     Update the elements of SIM and SIMI, and set the next X.
450                             temp = 0.0;
451                             for (int i = 1; i <= n; ++i) {
452                                 dx[i] = dxsign * dx[i];
453                                 sim[i][jdrop] = dx[i];
454                                 temp += simi[jdrop][i] * dx[i];
455                             }
456                             for (int k = 1; k <= n; ++k) {
457                                 simi[jdrop][k] /= temp;
458                             }
459 
460                             for (int j = 1; j <= n; ++j) {
461                                 if (j != jdrop) {
462                                     temp = dotProduct(part(row(simi, j), 1, n), part(dx, 1, n));
463                                     for (int k = 1; k <= n; ++k) {
464                                         simi[j][k] -= temp * simi[jdrop][k];
465                                     }
466                                 }
467                                 x[j] = sim[j][np] + dx[j];
468                             }
469                             continue L_40;
470                         }
471 
472                         //     Calculate DX = x(*)-x(0).
473                         //     Branch if the length of DX is less than 0.5*RHO.
474                         ifull = trstlp(n, m, a, con, rho, dx);
475                         if (!ifull) {
476                             temp = 0.0;
477                             for (int k = 1; k <= n; ++k) {
478                                 temp += dx[k] * dx[k];
479                             }
480                             if (temp < 0.25 * rho * rho) {
481                                 ibrnch = true;
482                                 break L_550;
483                             }
484                         }
485 
486                         //     Predict the change to F and the new maximum constraint violation if the
487                         //     variables are altered from x(0) to x(0) + DX.
488                         total = 0.0;
489                         double resnew = 0.0;
490                         con[mp] = 0.0;
491                         for (int k = 1; k <= mp; ++k) {
492                             total = con[k] - dotProduct(part(col(a, k), 1, n), part(dx, 1, n));
493                             if (k < mp) {
494                                 resnew = Math.max(resnew, total);
495                             }
496                         }
497 
498                         //     Increase PARMU if necessary and branch back if this change alters the
499                         //     optimal vertex. Otherwise PREREM and PREREC will be set to the predicted
500                         //     reductions in the merit function and the maximum constraint violation
501                         //     respectively.
502                         prerec = datmat[mpp][np] - resnew;
503                         double barmu = prerec > 0.0 ? total / prerec : 0.0;
504                         if (parmu < 1.5 * barmu) {
505                             parmu = 2.0 * barmu;
506                             if (iprint >= 2) {
507                                 System.out.format("%nIncrease in PARMU to %13.6f%n", parmu);
508                             }
509                             double phi = datmat[mp][np] + parmu * datmat[mpp][np];
510                             for (int j = 1; j <= n; ++j) {
511                                 temp = datmat[mp][j] + parmu * datmat[mpp][j];
512                                 if (temp < phi || (temp == phi && parmu == 0.0 && datmat[mpp][j] < datmat[mpp][np])) {
513                                     continue L_140;
514                                 }
515                             }
516                         }
517                         prerem = parmu * prerec - total;
518 
519                         //     Calculate the constraint and objective functions at x(*).
520                         //     Then find the actual reduction in the merit function.
521                         for (int k = 1; k <= n; ++k) {
522                             x[k] = sim[k][np] + dx[k];
523                         }
524                         ibrnch = true;
525                         continue L_40;
526                     }
527 
528                     skipVertexIdent = false;
529                     double vmold = datmat[mp][np] + parmu * datmat[mpp][np];
530                     double vmnew = f + parmu * resmax;
531                     double trured = vmold - vmnew;
532                     if (parmu == 0.0 && f == datmat[mp][np]) {
533                         prerem = prerec;
534                         trured = datmat[mpp][np] - resmax;
535                     }
536 
537                     //     Begin the operations that decide whether x(*) should replace one of the
538                     //     vertices of the current simplex, the change being mandatory if TRURED is
539                     //     positive. Firstly, JDROP is set to the index of the vertex that is to be
540                     //     replaced.
541                     double ratio = trured <= 0.0 ? 1.0 : 0.0;
542                     jdrop = 0;
543                     for (int j = 1; j <= n; ++j) {
544                         temp = Math.abs(dotProduct(part(row(simi, j), 1, n), part(dx, 1, n)));
545                         if (temp > ratio) {
546                             jdrop = j;
547                             ratio = temp;
548                         }
549                         sigbar[j] = temp * vsig[j];
550                     }
551 
552                     //     Calculate the value of ell.
553                     double edgmax = delta * rho;
554                     int l = 0;
555                     for (int j = 1; j <= n; ++j) {
556                         if (sigbar[j] >= parsig || sigbar[j] >= vsig[j]) {
557                             temp = veta[j];
558                             if (trured > 0.0) {
559                                 temp = 0.0;
560                                 for (int k = 1; k <= n; ++k) {
561                                     temp += Math.pow(dx[k] - sim[k][j], 2.0);
562                                 }
563                                 temp = Math.sqrt(temp);
564                             }
565                             if (temp > edgmax) {
566                                 l = j;
567                                 edgmax = temp;
568                             }
569                         }
570                     }
571                     if (l > 0) {
572                         jdrop = l;
573                     }
574 
575                     if (jdrop != 0) {
576                         //     Revise the simplex by updating the elements of SIM, SIMI and DATMAT.
577 
578                         temp = 0.0;
579                         for (int i = 1; i <= n; ++i) {
580                             sim[i][jdrop] = dx[i];
581                             temp += simi[jdrop][i] * dx[i];
582                         }
583                         for (int k = 1; k <= n; ++k) {
584                             simi[jdrop][k] /= temp;
585                         }
586                         for (int j = 1; j <= n; ++j) {
587                             if (j != jdrop) {
588                                 temp = dotProduct(part(row(simi, j), 1, n), part(dx, 1, n));
589                                 for (int k = 1; k <= n; ++k) {
590                                     simi[j][k] -= temp * simi[jdrop][k];
591                                 }
592                             }
593                         }
594                         for (int k = 1; k <= mpp; ++k) {
595                             datmat[k][jdrop] = con[k];
596                         }
597 
598                         //     Branch back for further iterations with the current RHO.
599                         if (trured > 0.0 && trured >= 0.1 * prerem) {
600                             continue L_140;
601                         }
602                     }
603                 } while (false);
604 
605                 if (!iflag) {
606                     ibrnch = false;
607                     continue L_140;
608                 }
609 
610                 if (rho <= rhoend) {
611                     status = CobylaExitStatus.NORMAL;
612                     break L_40;
613                 }
614 
615                 //     Otherwise reduce RHO if it is not at its least value and reset PARMU.
616                 double cmin = 0.0, cmax = 0.0;
617 
618                 rho *= 0.5;
619                 if (rho <= 1.5 * rhoend) {
620                     rho = rhoend;
621                 }
622                 if (parmu > 0.0) {
623                     double denom = 0.0;
624                     for (int k = 1; k <= mp; ++k) {
625                         cmin = datmat[k][np];
626                         cmax = cmin;
627                         for (int i = 1; i <= n; ++i) {
628                             cmin = Math.min(cmin, datmat[k][i]);
629                             cmax = Math.max(cmax, datmat[k][i]);
630                         }
631                         if (k <= m && cmin < 0.5 * cmax) {
632                             temp = Math.max(cmax, 0.0) - cmin;
633                             denom = denom <= 0.0 ? temp : Math.min(denom, temp);
634                         }
635                     }
636                     if (denom == 0.0) {
637                         parmu = 0.0;
638                     } else if (cmax - cmin < parmu * denom) {
639                         parmu = (cmax - cmin) / denom;
640                     }
641                 }
642                 if (iprint >= 2) {
643                     System.out.format("%nReduction in RHO to %1$13.6f  and PARMU = %2$13.6f%n", rho, parmu);
644                 }
645                 if (iprint == 2) {
646                     printIterationResult(nfvals, datmat[mp][np], datmat[mpp][np], col(sim, np), n);
647                 }
648 
649             } while (true);
650         } while (true);
651 
652         switch (status) {
653             case NORMAL:
654                 if (iprint >= 1) {
655                     System.out.format("%nNormal return from subroutine COBYLA%n");
656                 }
657                 if (ifull) {
658                     if (iprint >= 1) {
659                         printIterationResult(nfvals, f, resmax, x, n);
660                     }
661                     return status;
662                 }
663                 break;
664             case MAX_ITERATIONS_REACHED:
665                 if (iprint >= 1) {
666                     System.out.format("%nReturn from subroutine COBYLA because the MAXFUN limit has been reached.%n");
667                 }
668                 break;
669             case DIVERGING_ROUNDING_ERRORS:
670                 if (iprint >= 1) {
671                     System.out.format("%nReturn from subroutine COBYLA because rounding errors are becoming damaging.%n");
672                 }
673                 break;
674         }
675 
676         for (int k = 1; k <= n; ++k) {
677             x[k] = sim[k][np];
678         }
679         f = datmat[mp][np];
680         resmax = datmat[mpp][np];
681         if (iprint >= 1) {
682             printIterationResult(nfvals, f, resmax, x, n);
683         }
684 
685         return status;
686     }

688     private static boolean trstlp(int n, int m, double[][] a, double[] b, double rho, double[] dx) {
689         // N.B. Arguments Z, ZDOTA, VMULTC, SDIRN, DXNEW, VMULTD & IACT have been removed.
690 
691         //     This subroutine calculates an N-component vector DX by applying the
692         //     following two stages. In the first stage, DX is set to the shortest
693         //     vector that minimizes the greatest violation of the constraints
694         //       A(1,K)*DX(1)+A(2,K)*DX(2)+...+A(N,K)*DX(N) .GE. B(K), K = 2,3,...,M,
695         //     subject to the Euclidean length of DX being at most RHO. If its length is
696         //     strictly less than RHO, then we use the resultant freedom in DX to
697         //     minimize the objective function
698         //              -A(1,M+1)*DX(1) - A(2,M+1)*DX(2) - ... - A(N,M+1)*DX(N)
699         //     subject to no increase in any greatest constraint violation. This
700         //     notation allows the gradient of the objective function to be regarded as
701         //     the gradient of a constraint. Therefore the two stages are distinguished
702         //     by MCON .EQ. M and MCON .GT. M respectively. It is possible that a
703         //     degeneracy may prevent DX from attaining the target length RHO. Then the
704         //     value IFULL = 0 would be set, but usually IFULL = 1 on return.
705         //     In general NACT is the number of constraints in the active set and
706         //     IACT(1),...,IACT(NACT) are their indices, while the remainder of IACT
707         //     contains a permutation of the remaining constraint indices.  Further, Z
708         //     is an orthogonal matrix whose first NACT columns can be regarded as the
709         //     result of Gram-Schmidt applied to the active constraint gradients.  For
710         //     J = 1,2,...,NACT, the number ZDOTA(J) is the scalar product of the J-th
711         //     column of Z with the gradient of the J-th active constraint.  DX is the
712         //     current vector of variables and here the residuals of the active
713         //     constraints should be zero. Further, the active constraints have
714         //     nonnegative Lagrange multipliers that are held at the beginning of
715         //     VMULTC. The remainder of this vector holds the residuals of the inactive
716         //     constraints at DX, the ordering of the components of VMULTC being in
717         //     agreement with the permutation of the indices of the constraints that is
718         //     in IACT. All these residuals are nonnegative, which is achieved by the
719         //     shift RESMAX that makes the least residual zero.
720         //     Initialize Z and some other variables. The value of RESMAX will be
721         //     appropriate to DX = 0, while ICON will be the index of a most violated
722         //     constraint if RESMAX is positive. Usually during the first stage the
723         //     vector SDIRN gives a search direction that reduces all the active
724         //     constraint violations by one simultaneously.
725         // Local variables
726         double temp;
727 
728         int nactx = 0;
729         double resold = 0.0;
730 
731         double[][] z = new double[1 + n][1 + n];
732         double[] zdota = new double[2 + m];
733         double[] vmultc = new double[2 + m];
734         double[] sdirn = new double[1 + n];
735         double[] dxnew = new double[1 + n];
736         double[] vmultd = new double[2 + m];
737         int[] iact = new int[2 + m];
738 
739         int mcon = m;
740         int nact = 0;
741         for (int i = 1; i <= n; ++i) {
742             z[i][i] = 1.0;
743             dx[i] = 0.0;
744         }
745 
746         int icon = 0;
747         double resmax = 0.0;
748         if (m >= 1) {
749             for (int k = 1; k <= m; ++k) {
750                 if (b[k] > resmax) {
751                     resmax = b[k];
752                     icon = k;
753                 }
754             }
755             for (int k = 1; k <= m; ++k) {
756                 iact[k] = k;
757                 vmultc[k] = resmax - b[k];
758             }
759         }
760 
761         //     End the current stage of the calculation if 3 consecutive iterations
762         //     have either failed to reduce the best calculated value of the objective
763         //     function or to increase the number of active constraints since the best
764         //     value was calculated. This strategy prevents cycling, but there is a
765         //     remote possibility that it will cause premature termination.
766         boolean first = true;
767         do {
768             L_60:
769             do {
770                 if (!first || (first && resmax == 0.0)) {
771                     mcon = m + 1;
772                     icon = mcon;
773                     iact[mcon] = mcon;
774                     vmultc[mcon] = 0.0;
775                 }
776                 first = false;
777 
778                 double optold = 0.0;
779                 int icount = 0;
780 
781                 double step, stpful;
782 
783                 L_70:
784                 do {
785                     double optnew = mcon == m ? resmax : -dotProduct(part(dx, 1, n), part(col(a, mcon), 1, n));
786 
787                     if (icount == 0 || optnew < optold) {
788                         optold = optnew;
789                         nactx = nact;
790                         icount = 3;
791                     } else if (nact > nactx) {
792                         nactx = nact;
793                         icount = 3;
794                     } else {
795                         --icount;
796                     }
797                     if (icount == 0) {
798                         break L_60;
799                     }
800 
801                     //     If ICON exceeds NACT, then we add the constraint with index IACT(ICON) to
802                     //     the active set. Apply Givens rotations so that the last N-NACT-1 columns
803                     //     of Z are orthogonal to the gradient of the new constraint, a scalar
804                     //     product being set to zero if its nonzero value could be due to computer
805                     //     rounding errors. The array DXNEW is used for working space.
806                     double ratio;
807                     if (icon <= nact) {
808                         if (icon < nact) {
809                             //     Delete the constraint that has the index IACT(ICON) from the active set.
810 
811                             int isave = iact[icon];

818                                 temp = Math.sqrt(sp * sp + zdota[kp] * zdota[kp]);

820                                 double beta = sp / temp;

826                                     z[i][k] = temp;

829                                 vmultc[k] = vmultc[kp];
830                                 k = kp;
831                             } while (k < nact);

839                         //     change to the current vector of variables.

841                             //     Pick the next search direction of stage two.
842

856                             dxnew[k] = a[k][kk];
857                         }

860                         {
861                             int k = n;

863                                 double sp = 0.0;

868                                     spabs += Math.abs(temp);

895                             //     active set in order to make room for the new active constraint, because

897                             //     the old active constraints.  Set the elements of VMULTD to the multipliers

899                             //     constraint to be deleted, but branch if no suitable index can be found.

906                                     double zdvabs = 0.0;
907

909                                         temp = z[i][k] * dxnew[i];
910                                         zdotv += temp;
911                                         zdvabs += Math.abs(temp);
912                                     }
913                                     double acca = zdvabs + 0.1 * Math.abs(zdotv);
914                                     double accb = zdvabs + 0.2 * Math.abs(zdotv);
915                                     if (zdvabs < acca && acca < accb) {
916                                         temp = zdotv / zdota[k];

918                                             double tempa = vmultc[k] / temp;
919                                             if (ratio < 0.0 || tempa < ratio) {
920                                                 ratio = tempa;
921                                             }
922                                         }
923 
924                                         if (k >= 2) {
925                                             int kw = iact[k];
926                                             for (int i = 1; i <= n; ++i) {
927                                                 dxnew[i] -= temp * a[i][kw];
928                                             }
929                                         }
930                                         vmultd[k] = temp;
931                                     } else {
932                                         vmultd[k] = 0.0;
933                                     }
934                                 } while (--k > 0);
935                             }
936                             if (ratio < 0.0) {
937                                 break L_60;
938                             }

940                             //     Revise the Lagrange multipliers and reorder the active constraints so
941                             //     that the one to be replaced is at the end of the list. Also calculate the
942                             //     new value of ZDOTA(NACT) and branch if it is not acceptable.

945                             }

947                                 int isave = iact[icon];
948                                 double vsave = vmultc[icon];
949                                 int k = icon;
950                                 do {
951                                     int kp = k + 1;
952                                     int kw = iact[kp];

954                                     temp = Math.sqrt(sp * sp + zdota[kp] * zdota[kp]);
955                                     double alpha = zdota[kp] / temp;
956                                     double beta = sp / temp;
957                                     zdota[kp] = alpha * zdota[k];
958                                     zdota[k] = temp;
959                                     for (int i = 1; i <= n; ++i) {
960                                         temp = alpha * z[i][kp] + beta * z[i][k];
961                                         z[i][kp] = alpha * z[i][k] - beta * z[i][kp];
962                                         z[i][k] = temp;
963                                     }

966                                     k = kp;

968                                 iact[k] = isave;

972                             if (temp == 0.0) {

974                             }

976                             vmultc[icon] = 0.0;
977                             vmultc[nact] = ratio;
978                         } else {

980                             //     active set.

1031                     //     that reduces RESMAX to zero. The two statements below that include the
1032                     //     factor 1.0E-6 prevent some harmless underflows that occurred in a test
1033                     //     calculation. Further, we skip the step if it could be zero within a

1082                     //     can be attributed to computer rounding errors. First calculate the new

1097                                 zdotw = 0.0;
1098                             }

1100                             if (k >= 2) {
1101                                 int kk = iact[k];
1102                                 for (int i = 1; i <= n; ++i) {
1103                                     dxnew[i] -= vmultd[k] * a[i][kk];
1104                                 }
1105                             }
1106                         } while (k-- >= 2);
1107                         if (mcon > m) {
1108                             vmultd[nact] = Math.max(0.0, vmultd[nact]);
1109                         }
1110                     }
1111

1113                     for (int k = 1; k <= n; ++k) {
1114                         dxnew[k] = dx[k] + step * sdirn[k];
1115                     }
1116                     if (mcon > nact) {
1117                         int kl = nact + 1;
1118                         for (int k = kl; k <= mcon; ++k) {
1119                             int kk = iact[k];
1120                             double total = resmax - b[kk];
1121                             double sumabs = resmax + Math.abs(b[kk]);
1122                             for (int i = 1; i <= n; ++i) {
1123                                 temp = a[i][kk] * dxnew[i];
1124                                 total += temp;
1125                                 sumabs += Math.abs(temp);
1126                             }
1127                             double acca = sumabs + 0.1 * Math.abs(total);
1128                             double accb = sumabs + 0.2 * Math.abs(total);
1129                             if (sumabs >= acca || acca >= accb) {
1130                                 total = 0.0;
1131                             }
1132                             vmultd[k] = total;
1133                         }
1134                     }
1135 
1136                     //     Calculate the fraction of the step from DX to DXNEW that will be taken.
1137                     ratio = 1.0;
1138                     icon = 0;
1139                     for (int k = 1; k <= mcon; ++k) {
1140                         if (vmultd[k] < 0.0) {
1141                             temp = vmultc[k] / (vmultc[k] - vmultd[k]);
1142                             if (temp < ratio) {
